            switchTab('dashboardOverview');
        }

        function logout() {
            document.getElementById('dashboardView').classList.add('hidden');
            document.getElementById('landingView').classList.remove('hidden');
        }

        function switchTab(tabId) {
            // Hide all tabs
            const tabs = document.querySelectorAll('.tab-content');
            tabs.forEach(tab => tab.classList.add('hidden'));

            // Show target tab
            document.getElementById(tabId).classList.remove('hidden');

            // Reset buttons styling in sidebar
            const buttons = document.querySelectorAll('aside nav button');
            buttons.forEach(btn => {
                btn.className = "w-full flex items-center gap-3 px-4 py-3 rounded-xl text-sm font-medium text-slate-400 hover:bg-slate-900 hover:text-white transition";
            });

            // Highlight active button
            const activeBtn = document.getElementById(`btn-${tabId}`);
            if (activeBtn) {
                activeBtn.className = "w-full flex items-center gap-3 px-4 py-3 rounded-xl text-sm font-bold bg-indigo-600 text-white transition";
            }
        }

        // Search Engine
        function filterCourses() {
            const input = document.getElementById('courseSearch').value.toLowerCase();
            const cards = document.querySelectorAll('.course-card');

            cards.forEach(card => {
                const category = card.getAttribute('data-category');
                const title = card.querySelector('h3').innerText.toLowerCase();
                const text = card.querySelector('p').innerText.toLowerCase();

                if (category.includes(input) || title.includes(input) || text.includes(input)) {
                    card.style.display = "flex";
                } else {
                    card.style.display = "none";
                }
            });
        }
